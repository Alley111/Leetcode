[200. Number of Islands](https://leetcode.com/problems/number-of-islands/description/)

class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        
        rLen,cLen,cnt = len(grid),len(grid[0]),0
        def dfs(r, c):
            if(r<0 or r>=rLen or c<0 or c>=cLen or grid[r][c]!= "1"): return

            grid[r][c]="*"

            dfs(r+1,c)
            dfs(r-1,c)
            dfs(r,c+1)
            dfs(r,c-1)

            return 

        for r in range(rLen):
            for c in range(cLen):
                if grid[r][c]=="1":
                    dfs(r,c)
                    cnt+=1

        return cnt

class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        
        rLen,cLen,cnt = len(grid),len(grid[0]),0
        for r in range(rLen):
            for c in range(cLen):
                if grid[r][c]=="1":
                    # dfs(r,c)
                    cnt += 1
                    grid[r][c]="*"
                    queue = []
                    queue.append((r,c))
                    while queue:
                        r,c = queue.pop(0)
                        if r-1>=0 and grid[r-1][c]=="1":

                            queue.append((r,c+1))
                            grid[r][c+1]="*"

        return cnt
